#include <iostream>
#include <string>
using namespace std;
bool isPalindrome(string s){
    int l=0;
    int r=s.length()-1;
    while(l<r){
        if(s[l]!=s[r]){
            return false;
        }
        l++;
        r--;
    }
    return true;
}
int main() {
    string s="antioggna";
    bool ans=isPalindrome(s);
    if(ans) cout<<"Palindrome";
    else cout<<"Not a Palindrome";
    return 0;
}
