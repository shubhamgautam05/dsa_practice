// Valid Anagram
#include <iostream>
#include<vector>
#include<string>
using namespace std;
int main() {
    string s="shubhamu";
    string t="mahbushu";
    bool flag=false;
    vector<int> f1(26);
    vector<int> f2(26);
    for(char ch: s){
        f1[ch - 'a']++;
    }
    for(char ch: t){
        f2[ch - 'a']++;
    }
    if(f1==f2) flag=true;
    if(flag) cout<<"It is anagram";
    else cout<<"Not an anagram";
    return 0;
}
