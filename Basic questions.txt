Que-1: Print numbers from 1 to 10.

#include<iostream>
using namespace std;
int main(){
	for(int i=1;i<=10;i++){
		cout<<i<<" ";
	}
	return 0;
}

Que-2: Find sum of N numbers.

#include<iostream>
using namespace std;
int main(){
	int n;
	cin>>n;
	int sum=0;
	for(int i=1;i<=n;i++){
		sum+=i;
	}
	cout<<sum;
	return 0;
}


Que 3. Find the largest of 3 numbers.

#include<iostream>
using namespace std;
int main(){
	int a,b,c;
	cin>>a>>b>>c;
	if(a>b&&a>c) cout<<a;
	if(b>a&&b>c) cout<<b;
	else cout<<c;
	return 0;
}

Que 5. Reverse string.

#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main(){
	string s="hello";
	//simple
	//reverse(s.begin(),s.end());
	//cout<<s;
	// two pointer
 	int left=0;
 	int right=s.size()-1;
    while(left<right){	
 		swap(s[left],s[right]);
 		left++;
 		right--;
 	}
 	cout<<s;
