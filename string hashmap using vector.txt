
#include <iostream>
#include<vector>
#include<string>
using namespace std;
int main() {
    string str="shubham";
    vector<int> freq(26);
    for(char ch: str){
        freq[ch - 'a']++;
    }
    for(char ch: str){
        cout<<ch<<":"<<freq[ch - 'a']<<endl;
    }
    

    return 0;
}
