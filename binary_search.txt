#include<iostream>
#include<vector>
using namespace std;
int main(){
vector<int> v={1,2,3,4,5,6};
bool flag=false;
int target=5;
int low=0;
int high=v.size()-1;
while(low<=high){
	int mid=low+((high-low)/2);
	if(v[mid]==target){
		flag=true;
		break;
            }
	else if(v[mid]>target) high=mid-1;
	else low=mid+1;
}
if(flag) cout<<"element found";
else cout<<"element not found";
return 0;
}
