#include<iostream>
#include<vector>
using namespace std;
int floorOccurence(vector<int>& v, int target){
	int l=0;
	int h=v.size()-1;
	int ans=-1;
	while(l<=h){
	int mid=l+((h-l)/2);
	if(v[mid]==target){
	    ans=mid;
	    h=mid-1;
	}
	else if(v[mid]>target) h=mid-1;
	else l=mid+1;
}
if(v[ans]==v[0]){
    return v[ans];
}
ans--;
return v[ans];
}
int cielOccurence(vector<int>& v, int target){
	int l=0;
	int h=v.size()-1;
	int ans=-1;
	while(l<=h){
	int mid=l+((h-l)/2);
	if(v[mid]==target){
	    ans=mid;
	    l=mid+1;
	}
	else if(v[mid]>target) h=mid-1;
	else l=mid+1;
}
if(v[ans]==v[v.size()-1]){
    return v[ans];
}
ans++;
return v[ans];
}
int main(){
	vector<int> v={1,2,4,6,7,9,9,9,9,9,10};
	int target=1;
	int ans = floorOccurence(v,target);
	cout<<"Floor Occurence: "<<ans<<endl;
	int ciel= cielOccurence(v,target);
	cout<<"Ciel Occurence: "<<ciel<<endl;
return 0;
}
