#include<iostream>
#include<vector>
using namespace std;
int firstOccurence(vector<int>& v, int target){
	int l=0;
	int h=v.size()-1;
	int ans=-1;
	while(l<=h){
	int mid=l+((h-l)/2);
	if(v[mid]==target){
	    ans=mid;
	    h=mid-1;
	}
	else if(v[mid]>target) h=mid-1;
	else l=mid+1;
}
return ans;
}
int main(){
	vector<int> v={1,2,4,6,8,9,9,9,9,9,10};
	int target=9;
	int ans = firstOccurence(v,target);
	cout<<ans;
return 0;
}
