#include<iostream>
#include<vector>
using namespace std;
void ReverseArray(vector<int>& arr){
	int left=0;
	int right=arr.size()-1;
	while(left<right){
		//swap(arr[left],arr[right]);
		arr[left]^=arr[right];
		arr[right]^=arr[left];
		arr[left]^=arr[right];
		left++;
		right--;
}
}
int main(){
	vector<int>arr={1,2,3,4,5};
	ReverseArray(arr);
	for(int i:arr){
		cout<<i<<" ";
	}
return 0;
}
