#include<iostream>
#include<vector>
using namespace std;
    void removeDuplicates(vector<int>& nums) {
        int i=0;
        for(int j=1;j<nums.size();j++){
            if(nums[i]!=nums[j]){
                nums[i+1]=nums[j];
                i++;
            }
        }
        while(i<nums.size()-1){
            nums[i+1]=0;
            i++;
        }
    }
int main(){
	vector<int> v={1,1,2,2,3,3,6,6,7};
	removeDuplicates(v);
	for(int i: v){
		cout<<i<<" ";
	}	

return 0;
}
